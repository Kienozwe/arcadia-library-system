📄 SQL DDL

-- Arcadia Library Management System -- MySQL 8.0+ compatible schema CREATE DATABASE arcadia_library CHARACTER SET utf8mb4; USE arcadia_library; -- Users table (students + admin) CREATE TABLE users ( id INT AUTO_INCREMENT PRIMARY KEY, name VARCHAR(150) NOT NULL, email VARCHAR(255) NOT NULL UNIQUE, password_hash VARCHAR(255) NOT NULL, role ENUM('student','admin') DEFAULT 'student', student_id VARCHAR(50), course VARCHAR(150), year_level VARCHAR(30), address TEXT, created_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ); -- Books table CREATE TABLE books ( id INT AUTO_INCREMENT PRIMARY KEY, title VARCHAR(255) NOT NULL, author VARCHAR(150) NOT NULL, category VARCHAR(80), isbn VARCHAR(20) UNIQUE, description TEXT, cover_url VARCHAR(500), total_copies INT DEFAULT 1, available_copies INT DEFAULT 1, added_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP ); -- Borrow requests / transactions CREATE TABLE borrow_requests ( id INT AUTO_INCREMENT PRIMARY KEY, book_id INT NOT NULL, user_id INT NOT NULL, student_name VARCHAR(150), student_id_no VARCHAR(50), year_level VARCHAR(30), course VARCHAR(150), address TEXT, status ENUM('pending','approved','ready','rejected','returned') DEFAULT 'pending', requested_at TIMESTAMP DEFAULT CURRENT_TIMESTAMP, due_date DATE, returned_at DATE, admin_note TEXT, FOREIGN KEY (book_id) REFERENCES books(id) ON DELETE CASCADE, FOREIGN KEY (user_id) REFERENCES users(id) ON DELETE CASCADE ); -- Indexes for performance CREATE INDEX idx_requests_status ON borrow_requests(status); CREATE INDEX idx_requests_user ON borrow_requests(user_id); CREATE INDEX idx_books_category ON books(category);

// api/borrow.php — Example PHP endpoint header('Content-Type: application/json'); require_once 'db_connect.php'; // PDO connection if ($_SERVER['REQUEST_METHOD'] === 'POST') { $data = json_decode(file_get_contents('php://input'), true); $stmt = $pdo->prepare( "INSERT INTO borrow_requests (book_id, user_id, student_name, student_id_no, year_level, course, address) VALUES (:book_id, :user_id, :name, :sid, :year, :course, :addr)" ); $stmt->execute([ 'book_id' => $data['bookId'], 'user_id' => $data['userId'], 'name' => $data['studentName'], 'sid' => $data['studentId'], 'year' => $data['yearLevel'], 'course' => $data['course'], 'addr' => $data['address'], ]); // Decrement available_copies $pdo->prepare("UPDATE books SET available_copies = available_copies - 1 WHERE id = :id AND available_copies > 0") ->execute(['id' => $data['bookId']]); echo json_encode(['success' => true, 'id' => $pdo->lastInsertId()]); }
